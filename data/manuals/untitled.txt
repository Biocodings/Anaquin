cat A.V.5.bed A.V.13.bed > combined.bed

samtools view -b -L combined.bed sample_alignments.bam > SV_sample_regions.bam

anaquin VarTrim -o 6.4.10 -rbed combined.bed -usequin sequin_alignments.bam \
     | samtools view -bS - | samtools sort -o SV_sequin_trimmed.bam

anaquin VarSubsample -o 6.4.10 -method mean -rbed combined.bed \
    -usample SV_sample_regions.bam -usequin SV_sequin_trimmed.bam -edge 500 | \
    samtools view -bS - | samtools sort -o SV_sequin_sampled.bam

# Realign for sample derived reads

speedseq realign -t 20 -R '@RG\tID:sequins\tSM:sequins\tPL:illumina' -o 6.4.10_ReAlign_1 hg38.fa sample_alignments.bam
speedseq sv -B sample_alignments.bam -S sample_splitters.bam -D sample_discordants.bam -R hg38.fa -t 20 -o 6.4.10_1

speedseq realign -t 20 -R '@RG\tID:sequins\tSM:sequins\tPL:illumina' -o 6.4.10_ReAlign_2 hg38.fa SV_sequin_sampled.bam
speedseq sv -B SV_sequin_sampled.bam -S sequin_splitters.bam -D sequin_discordants.bam -R hg38.fa -t 20 -o 6.4.10_2
